load()



Loads the locales in a given file.
             		                            
  load: (locale, file, cb) ->
    fs.exists requirejs.toUrl('i18n/' + file), (exists) ->
      if exists
        requirejs
          locale: locale
          context: locale
          baseUrl: requirejs.toUrl('./')
        , ['i18n!i18n/' + file], cb
      else
        cb {}
